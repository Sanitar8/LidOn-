DELIMITER $$

CREATE TRIGGER trg_user_prof_limit
BEFORE INSERT ON user_professions
FOR EACH ROW
BEGIN
  DECLARE cnt INT;
  SELECT COUNT(*) INTO cnt FROM user_professions WHERE user_id = NEW.user_id;
  IF cnt >= 3 THEN
    SIGNAL SQLSTATE '45000'
      SET MESSAGE_TEXT = 'Можно выбрать максимум 3 профессии';
  END IF;
END$$

DELIMITER ;
