CREATE TABLE IF NOT EXISTS user_professions (
  user_id INT UNSIGNED NOT NULL,
  profession_id INT UNSIGNED NOT NULL,
  is_primary TINYINT(1) NOT NULL DEFAULT 0,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,

  PRIMARY KEY (user_id, profession_id),

  KEY idx_up_user (user_id),
  KEY idx_up_prof (profession_id),
  KEY idx_up_primary (user_id, is_primary),

  CONSTRAINT fk_up_user
    FOREIGN KEY (user_id) REFERENCES users(id)
    ON DELETE CASCADE,

  CONSTRAINT fk_up_prof
    FOREIGN KEY (profession_id) REFERENCES professions(id)
    ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
